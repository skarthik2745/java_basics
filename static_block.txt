// Static Blocks in Java
class BlockTest {
    // First static block of BlockTest class
    static {
        System.out.println("BlockTest-1");
    }

    // Second static block of BlockTest class
    static {
        System.out.println("BlockTest-2");
    }
}

public class staticBlocks {
    // Static block inside staticBlocks class
    static {
        System.out.println("Block-1");
    }

    public static void main(String args[]) {
        // When we create an object of BlockTest,
        // its static blocks are executed first (only once).
        BlockTest o = new BlockTest();

        // Then the main method continues
        System.out.println("Main");
    }

    // Another static block (after main)
    static {
        System.out.println("Block-2");
    }
}
