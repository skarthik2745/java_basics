class Box {
    int length;
    int breadth;
    int height;

    // Constructor
    Box(int l, int b, int h) {
        length = l;
        breadth = b;
        height = h;
    }

    // Method to check equality of two Box objects
    boolean isEqual(Box b) {
        return (length == b.length && breadth == b.breadth && height == b.height);
    }

    // Method that returns a new Box object with doubled dimensions
    Box doubleBox() {
        return new Box(2 * length, 2 * breadth, 2 * height);
    }

    // Overriding toString() for easy printing
    public String toString() {
        return "Box(" + length + ", " + breadth + ", " + height + ")";
    }
}

public class Main {

    

    public static void main(String[] args) {
        
        Box b1 = new Box(5, 3, 4);
        Box b2 = new Box(5, 3, 4);

        // Using the doubleBox() method
        Box b3 = b1.doubleBox();

        System.out.println("b1 equals b2? " + b1.isEqual(b2)); // true
        System.out.println("b2 length: " + b2.length);         // 5
        System.out.println("b3 length: " + b3.length);         // 10

        // Printing full objects
        System.out.println("b1: " + b1);
        System.out.println("b2: " + b2);
        System.out.println("b3: " + b3);
    }
}
